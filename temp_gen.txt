    20	    private fun createBarcodeBitmap(contents: String, format: BarcodeFormat, width: Int, height: Int): Bitmap? {
    21	        if (contents.isEmpty()) return null
    22	        return try {
    23	            val writer = MultiFormatWriter()
    24	            val matrix = writer.encode(contents, format, width, height)
    25	            val bmp = Bitmap.createBitmap(width, height, Bitmap.Config.ARGB_8888)
    26	            for (x in 0 until width) {
    27	                for (y in 0 until height) {
    28	                    bmp.setPixel(x, y, if (matrix.get(x, y)) Color.BLACK else Color.WHITE)
    29	                }
    30	            }
    31	            bmp
    32	        } catch (e: Exception) {
    33	            e.printStackTrace()
    34	            null
    35	        }
    36	    }

   473	
   474	        // --- Left Column: Set Codes ---
   475	        val setSets = listOf("A", "B", "C", "D", "E", "F", "G", "H", "I", "J")
   476	        val setBubbleRadius = 13f
   477	        val setGridStartX = boxLeft + 105f
   478	        val setGridStartY = boxTop + 80f
   479	        val setRowHeight = 60f
   480	        
   481	        for (i in setSets.indices) {

   484	            canvas.drawCircle(setGridStartX, cy, setBubbleRadius, thinStroke)

   486	        }
   487	
   488	        // --- Right Column: Objective Questions Answers ---
   489	        val numCols = 5
   490	        val numQPerCol = 20
   491	        val rightBoxWidth = boxRight - splitX
   492	        val colWidth = rightBoxWidth / numCols
   493	        val bubbleRadius = 10f
   494	        val optSpacing = 21f
   495	        val qRowHeight = 39f
   496	        val optLetters = listOf("A", "B", "C", "D")
   497	        
   498	        var currentQ = 1
   499	        
   500	        for (col in 0 until numCols) {

   502	            val endX = startX + colWidth
   503	            val qNumX = startX + 18f
   504	            val bubblesStartX = startX + 46f
   505	            
   506	            // Draw column separator if not first column
   507	            if (col > 0) {
   508	                canvas.drawLine(startX, boxTop + 40f, startX, boxBottom, blackStroke)
   509	            }
   510	            
   511	            // Draw Column Header

   516	            for (opt in 0..3) {
   517	                canvas.drawText(optLetters[opt], bubblesStartX + opt * optSpacing, headerTop + 20f, headerTextPaint)
   518	            }
   519	            
   520	            for (row in 0 until numQPerCol) {
   521	                val rowTop = headerBottom + row * qRowHeight
   522	                val rowBottom = rowTop + qRowHeight
   523	                val cy = rowTop + qRowHeight / 2f
   524	                
   525	                val qString = currentQ.toString()
   526

   529	                
   530	                // Draw Options

   533	                    canvas.drawCircle(cx, cy, bubbleRadius, thinStroke)
   534	                    canvas.drawText(optLetters[opt], cx, cy + 4f, bubbleTextPaint)
   535	                }
   536	                currentQ++
   537	            }
   538	        }
